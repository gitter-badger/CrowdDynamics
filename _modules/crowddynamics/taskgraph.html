        """Evaluate nodes recursively, child nodes first (bottom-up)."""
        for node in self._children:
            node.evaluate()
        self.update()

        """Add new child node.

            node: TaskNode

    def __iadd__(self, other):
        """Syntactic sugar for adding new child nodes.

        Args:
            other: TaskNode

        Returns:
            TaskNode: self

        self.add(other)
